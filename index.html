# Dynamic Re-ranking with Feedback
def rerank_with_feedback(scores, ideal_idx):
    # Get ideal candidate embedding
    ideal_emb = embeddings[ideal_idx]
    new_scores = {}

    for idx, emb in embeddings.items():
        sim_to_ideal = cosine_similarity(ideal_emb, emb)
        # Weighted: 60% original, 40% similarity to ideal
        combined = 0.6 * scores[idx] + 0.4 * sim_to_ideal
        new_scores[idx] = combined

    return sorted(new_scores.items(), key=lambda x: x[1], reverse=True)
# NDCG & MRR Ranking Metrics
def calculate_ndcg(true_rel, pred_scores, k=10):
    # Ideal DCG: sort by true relevance
    ideal_dcg = sum((2**rel - 1) / np.log2(i + 2)
                    for i, rel in
                    enumerate(sorted(true_rel, reverse=True)[:k]))

    # Actual DCG: based on predictions
    pred_rank = np.argsort(-pred_scores)[:k]
    actual_dcg = sum((2**true_rel[i] - 1) / np.log2(j + 2)
                     for j, i in enumerate(pred_rank))

    return actual_dcg / ideal_dcg if ideal_dcg > 0 else 0

# Bollinger Bands & Technical Indicators
def calculate_bollinger_bands(prices, window=20, std_dev=2):
    sma = prices.rolling(window=window).mean()
    std = prices.rolling(window=window).std()
    
    upper_band = sma + (std * std_dev)
    lower_band = sma - (std * std_dev)
    
    return upper_band, sma, lower_band

# Calculate RSI
def calculate_rsi(prices, period=14):
    delta = prices.diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=period).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=period).mean()
    
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    return rsi
# Generate Trading Signals
def generate_signals(data):
    upper, middle, lower = calculate_bollinger_bands(data['Close'])
    rsi = calculate_rsi(data['Close'])
    
    signals = []
    for i in range(1, len(data)):
        price = data['Close'].iloc[i]
        
        if price < lower.iloc[i] and rsi.iloc[i] < 30:
            signals.append('BUY')
        elif price > upper.iloc[i] and rsi.iloc[i] > 70:
            signals.append('SELL')
        else:
            signals.append('HOLD')
    
    return signals
# Backtest Trading Strategy
def backtest_strategy(data, signals):
    capital = 10000
    position = False
    entry_price = 0
    returns = []
    
    for i, signal in enumerate(signals):
        price = data['Close'].iloc[i]
        
        if signal == 'BUY' and not position:
            position = True
            entry_price = price
        elif signal == 'SELL' and position:
            capital *= (1 + (price - entry_price) / entry_price)
            position = False
            returns.append(capital)

    return capital, returns

def wave_trend(hlc3, period1=9, period2=12):
    esa = ta.EMA(hlc3, period1)
    de = ta.EMA(abs(hlc3 - esa), period1)

    lows = np.array([float(x[3]) for x in klines])

    if wt1[-1] > wt2[-1] and wt2[-1] <= -20:

    elif wt1[-1] < wt2[-1] and wt2[-1] >= 20:

# ATR-Based Stop Loss Calculation
def calculate_atr(symbol, period=14):
    klines = client.get_klines(symbol=symbol, interval=Client.KLINE_INTERVAL_15MINUTE, limit=period)
    highs = np.array([float(x[2]) for x in klines])
    lows = np.array([float(x[3]) for x in klines])
    closes = np.array([float(x[4]) for x in klines])
    
    tr = np.maximum(highs - lows, np.maximum(abs(highs - closes[:-1]), abs(lows - closes[:-1])))
    atr = np.mean(tr)
    return atr

def set_stop_loss(entry_price, symbol):
    atr = calculate_atr(symbol)
    stop_loss = entry_price - (2 * atr)
    trailing_stop = entry_price * (1 - trail)
